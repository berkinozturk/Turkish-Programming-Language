*** Listeler ve olduğu sürece döngüsü ***
filmler = liste ("Soul", "Sindirella", "Mulan", "Mega Zeka", "Rapunzel", "Moana", "Zootopia", "Arabalar");

a=1;
(a==1) olduğu sürece{
    yaz("yapmak istediğiniz işlemi yazınız: yazdir-ekle-sil-kapat");
    komut = kelimeyi oku;
    eğer (komut == "yazdir") ise {
        yaz(filmler);
    }
    ya da (komut == "ekle") ise {
        yaz("Eklemek istediğiniz filmin ismini yazın: ");
        eklenecek = kelimeyi oku;
        filmler listesine eklenecek ekle;
        yaz(filmler);
    }
    ya da (komut == "sil") ise{
        yaz("Silmek istediğiniz filmin ismini yazın: ");
        silinecek = kelimeyi oku;
        siraNo = filmler listesindeki silinecek elemanının sırası;
        filmler listesindeki siraNo. elemanı sil;
        yaz(filmler);

    }
    ya da (komut == "kapat") ise{
        yaz("Programdan çıkış yapılıyor...");
        a=0; ***burada kır düzgün çalışmadığı için böyle yaptım ***
    }
     değil ise {
        yaz("geçersiz bir komut girdiniz");
    };
};
